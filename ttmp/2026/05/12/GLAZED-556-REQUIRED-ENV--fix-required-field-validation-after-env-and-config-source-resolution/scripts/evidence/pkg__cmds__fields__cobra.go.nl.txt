    13	// GenerateUseString creates a string representation of the 'Use' field for a given cobra command and a list of field definitions. The first word of the existing 'Use' field is treated as the verb for the command.
    14	// The resulting string briefly describes how to use the command respecting the following conventions:
    15	//   - Required fields are enclosed in '<>'.
    16	//   - Optional fields are enclosed in '[]'.
    17	//   - Optional fields that accept multiple input (TypeStringList or TypeIntegerList) are followed by '...'.
    18	//   - If a field has a default value, it is specified after field name like 'field (default: value)'.
    19	//
    20	// For example:
    21	//   - If there is a required field 'name', and an optional field 'age' with a default value of '30', the resulting string will be: 'verb <name> [age (default: 30)]'.
    22	//   - If there is a required field 'name', and an optional field 'colors' of type TypeStringList, the resulting Use string will be: 'verb <name> [colors...]'
    23	func GenerateUseString(cmdName string, pds *Definitions) string {
    24		fields := strings.Fields(cmdName)
    25		if len(fields) == 0 {
    26			return ""
    27		}
    28		verb := fields[0]
    29		useStr := verb
    30		var defaultValueStr string
    31	
    32		pds.ForEach(func(arg *Definition) {
    33			if !arg.IsArgument {
    34				return
    35			}
    36			defaultValueStr = ""
    37			if arg.Default != nil {
    38				defaultValueStr = fmt.Sprintf(" (default: %v)", *arg.Default)
    39			}
    40			left, right := "[", "]"
    41			if arg.Required {
    42				left, right = "<", ">"
    43			}
    44			if arg.Type == TypeStringList || arg.Type == TypeIntegerList {
    45				useStr += " " + left + arg.Name + "..." + defaultValueStr + right
    46			} else {
    47				useStr += " " + left + arg.Name + defaultValueStr + right
    48			}
    49		})
    50	
    51		return useStr
    52	}

    54	// addArgumentsToCobraCommand adds each Definition from `arguments` as positional arguments to the provided `cmd` cobra command.
    55	// Argument ordering, optionality, and multiplicity constraints are respected:
    56	//   - Required arguments (argument.Required == true) should come before the optional.
    57	//   - Only one list argument (either TypeStringList or TypeIntegerList) is allowed and it should be the last one.
    58	//
    59	// Any violation of these conditions yields an error.
    60	// This function processes each argument, checks their default values for validity, which if invalid,
    61	// triggers an error return.
    62	//
    63	// It computes the minimum and maximum number of arguments the command can take based on the required, optional,
    64	// and list arguments.
    65	// If everything is successful, it assigns an argument validator (either MinimumNArgs or RangeArgs)
    66	// to the cobra command's Args attribute.
    67	func (pds *Definitions) addArgumentsToCobraCommand(cmd *cobra.Command) error {
    68		minArgs := 0
    69		// -1 signifies unbounded
    70		maxArgs := 0
    71		hadOptional := false
    72	
    73		if pds.Len() == 0 {
    74			return nil
    75		}
    76	
    77		err := pds.ForEachE(func(argument *Definition) error {
    78			if !argument.IsArgument {
    79				return nil
    80			}
    81			if maxArgs == -1 {
    82				// already handling unbounded arguments
    83				return errors.Errorf("Cannot handle more than one unbounded argument, but found %s", argument.Name)
    84			}
    85			_, err := argument.CheckDefaultValueValidity()
    86			if err != nil {
    87				return errors.Wrapf(err, "Invalid default value for argument %s", argument.Name)
    88			}
    89	
    90			if argument.Required {
    91				if hadOptional {
    92					return errors.Errorf("Cannot handle required argument %s after optional argument", argument.Name)
    93				}
    94				minArgs++
    95			} else {
    96				hadOptional = true
    97			}
    98			maxArgs++
    99			if argument.Type.IsList() {
   100				maxArgs = -1
   101			}
   102	
   103			return nil
   104		})
   105	
   106		if err != nil {
   107			return err
   108		}
   109	
   110		cmd.Args = cobra.MinimumNArgs(minArgs)
   111		if maxArgs != -1 {
   112			cmd.Args = cobra.RangeArgs(minArgs, maxArgs)
   113		}
   114	
   115		cmd.Use = GenerateUseString(cmd.Use, pds)
   116	
   117		return nil
   118	}

   120	// AddFieldsToCobraCommand takes the fields from a CommandDescription and converts them
   121	// to cobra flags, before adding them to the Fields() of a the passed cobra command.
   122	func (pds *Definitions) AddFieldsToCobraCommand(
   123		cmd *cobra.Command,
   124		prefix string,
   125	) error {
   126		flagSet := cmd.Flags()
   127	
   128		err := pds.GetArguments().addArgumentsToCobraCommand(cmd)
   129		if err != nil {
   130			return err
   131		}
   132	
   133		err = pds.GetFlags().ForEachE(func(field *Definition) error {
   134			_, err := field.CheckDefaultValueValidity()
   135			if err != nil {
   136				return errors.Wrapf(err, "Invalid default value for argument %s", field.Name)
   137			}
   138	
   139			flagName := prefix + field.Name
   140			// replace _ with -
   141			flagName = strings.ReplaceAll(flagName, "_", "-")
   142			shortFlag := field.ShortFlag
   143			if prefix != "" {
   144				// we don't allow shortflags if a prefix was given
   145				shortFlag = ""
   146			}
   147			ok := false
   148	
   149			f := flagSet.Lookup(flagName)
   150			if f != nil {
   151				return errors.Errorf("Flag '%s' (usage: %s) already exists", flagName, f.Usage)
   152			}
   153	
   154			helpText := field.Help
   155			helpText = fmt.Sprintf("%s - <%s>", helpText, field.Type)
   156	
   157			switch field.Type {
   158			case TypeStringListFromFile,
   159				TypeStringFromFile,
   160				TypeObjectFromFile,
   161				TypeObjectListFromFile,
   162				TypeFile:
   163				defaultValue := ""
   164	
   165				if field.ShortFlag != "" {
   166					flagSet.StringP(flagName, shortFlag, defaultValue, helpText)
   167				} else {
   168					flagSet.String(flagName, defaultValue, helpText)
   169				}
   170	
   171			case TypeStringListFromFiles,
   172				TypeStringFromFiles,
   173				TypeObjectListFromFiles,
   174				TypeFileList:
   175				defaultValue := []string{}
   176	
   177				if field.ShortFlag != "" {
   178					flagSet.StringSliceP(flagName, shortFlag, defaultValue, helpText)
   179				} else {
   180					flagSet.StringSlice(flagName, defaultValue, helpText)
   181				}
   182	
   183			case TypeString, TypeSecret:
   184				defaultValue := ""
   185	
   186				if field.Default != nil {
   187					defaultValue, err = cast.ToString(*field.Default)
   188					if err != nil {
   189						return errors.Errorf("Default value for field %s is not a string: %v", field.Name, *field.Default)
   190					}
   191				}
   192	
   193				if field.ShortFlag != "" {
   194					flagSet.StringP(flagName, shortFlag, defaultValue, helpText)
   195				} else {
   196					flagSet.String(flagName, defaultValue, helpText)
   197				}
   198			case TypeInteger:
   199				defaultValue := 0
   200	
   201				if field.Default != nil {
   202					defaultValue, ok = cast.CastNumberInterfaceToInt[int](*field.Default)
   203					if !ok {
   204						return errors.Errorf("Default value for field %s is not an integer: %v", field.Name, *field.Default)
   205					}
   206				}
   207	
   208				if field.ShortFlag != "" {
   209					flagSet.IntP(flagName, shortFlag, defaultValue, helpText)
   210				} else {
   211					flagSet.Int(flagName, defaultValue, helpText)
   212				}
   213	
   214			case TypeFloat:
   215				defaultValue := 0.0
   216	
   217				if field.Default != nil {
   218					defaultValue, ok = cast.CastFloatInterfaceToFloat[float64](*field.Default)
   219					if !ok {
   220						return errors.Errorf("Default value for field %s is not a float: %v", field.Name, *field.Default)
   221					}
   222				}
   223	
   224				if field.ShortFlag != "" {
   225					flagSet.Float64P(flagName, shortFlag, defaultValue, helpText)
   226				} else {
   227					flagSet.Float64(flagName, defaultValue, helpText)
   228				}
   229	
   230			case TypeBool:
   231				defaultValue := false
   232	
   233				if field.Default != nil {
   234					defaultValue, ok = (*field.Default).(bool)
   235					if !ok {
   236						return errors.Errorf("Default value for field %s is not a bool: %v", field.Name, *field.Default)
   237					}
   238				}
   239	
   240				if field.ShortFlag != "" {
   241					flagSet.BoolP(flagName, shortFlag, defaultValue, helpText)
   242				} else {
   243					flagSet.Bool(flagName, defaultValue, helpText)
   244				}
   245	
   246			case TypeDate:
   247				defaultValue := ""
   248	
   249				if field.Default != nil {
   250					switch v_ := (*field.Default).(type) {
   251					case string:
   252						_, err2 := ParseDate(v_)
   253						if err2 != nil {
   254							return err2
   255						}
   256						defaultValue = v_
   257					case time.Time:
   258						// nothing to do
   259						defaultValue = v_.Format("2006-01-02")
   260					default:
   261						return errors.Errorf("Default value for field %s is not a valid date: %v", field.Name, *field.Default)
   262					}
   263				}
   264	
   265				if field.ShortFlag != "" {
   266					flagSet.StringP(flagName, shortFlag, defaultValue, helpText)
   267				} else {
   268					flagSet.String(flagName, defaultValue, helpText)
   269				}
   270	
   271			case TypeStringList, TypeChoiceList:
   272				var defaultValue []string
   273	
   274				if field.Default != nil {
   275					stringList, ok := (*field.Default).([]string)
   276					if !ok {
   277						defaultValue, ok := (*field.Default).([]interface{})
   278						if !ok {
   279							return errors.Errorf("Default value for field %s is not a string list: %v", field.Name, *field.Default)
   280						}
   281	
   282						// convert to string list
   283						stringList, ok = cast.CastList[string, interface{}](defaultValue)
   284						if !ok {
   285							return errors.Errorf("Default value for field %s is not a string list: %v", field.Name, *field.Default)
   286						}
   287					}
   288	
   289					defaultValue = stringList
   290				}
   291	
   292				if field.ShortFlag != "" {
   293					flagSet.StringSliceP(flagName, shortFlag, defaultValue, helpText)
   294				} else {
   295					flagSet.StringSlice(flagName, defaultValue, helpText)
   296				}
   297	
   298			case TypeKeyValue:
   299				var defaultValue []string
   300	
   301				if field.Default != nil {
   302					stringMap, ok := (*field.Default).(map[string]string)
   303					if !ok {
   304						defaultValue, ok := (*field.Default).(map[string]interface{})
   305						if !ok {
   306							return errors.Errorf("Default value for field %s is not a string list: %v", field.Name, *field.Default)
   307						}
   308	
   309						stringMap = make(map[string]string)
   310						for k, v := range defaultValue {
   311							stringMap[k] = fmt.Sprintf("%v", v)
   312						}
   313					}
   314	
   315					stringList := make([]string, 0)
   316					for k, v := range stringMap {
   317						// TODO(manuel, 2023-02-11) This is fixed to : but should be configurable
   318						// See https://github.com/go-go-golems/glazed/issues/129
   319						stringList = append(stringList, fmt.Sprintf("%s:%s", k, v))
   320					}
   321	
   322					defaultValue = stringList
   323				}
   324				if err != nil {
   325					return errors.Wrapf(err, "Could not convert default value for field %s to string list: %v", field.Name, *field.Default)
   326				}
   327	
   328				if field.ShortFlag != "" {
   329					flagSet.StringSliceP(flagName, shortFlag, defaultValue, helpText)
   330				} else {
   331					flagSet.StringSlice(flagName, defaultValue, helpText)
   332				}
   333	
   334			case TypeIntegerList:
   335				var defaultValue []int
   336				if field.Default != nil {
   337					defaultValue, ok = cast.CastInterfaceToIntList[int](*field.Default)
   338					if !ok {
   339						return errors.Errorf("Default value for field %s is not an integer list: %v", field.Name, *field.Default)
   340					}
   341				}
   342	
   343				if field.ShortFlag != "" {
   344					flagSet.IntSliceP(flagName, shortFlag, defaultValue, helpText)
   345				} else {
   346					flagSet.IntSlice(flagName, defaultValue, helpText)
   347				}
   348	
   349			case TypeFloatList:
   350				var defaultValue []float64
   351				if field.Default != nil {
   352					defaultValue, ok = cast.CastInterfaceToFloatList[float64](*field.Default)
   353					if !ok {
   354						return errors.Errorf("Default value for field %s is not a float list: %v", field.Name, *field.Default)
   355					}
   356				}
   357				if field.ShortFlag != "" {
   358					flagSet.Float64SliceP(flagName, shortFlag, defaultValue, helpText)
   359				} else {
   360					flagSet.Float64Slice(flagName, defaultValue, helpText)
   361				}
   362	
   363			case TypeChoice:
   364				defaultValue := ""
   365	
   366				if field.Default != nil {
   367					defaultValue, err = cast.ToString(*field.Default)
   368					if err != nil {
   369						return errors.Errorf("Default value for field %s is not a string: %v", field.Name, *field.Default)
   370					}
   371				}
   372	
   373				choiceString := strings.Join(field.Choices, ",")
   374	
   375				if field.ShortFlag != "" {
   376					flagSet.StringP(flagName, shortFlag, defaultValue, fmt.Sprintf("%s (%s)", helpText, choiceString))
   377				} else {
   378					flagSet.String(flagName, defaultValue, fmt.Sprintf("%s (%s)", helpText, choiceString))
   379				}
   380	
   381			default:
   382				return errors.Errorf("Unknown field type for field %s: %s", field.Name, field.Type)
   383			}
   384	
   385			if field.Type.IsSensitive() && field.Default != nil {
   386				if f := flagSet.Lookup(flagName); f != nil && f.DefValue != "" {
   387					f.DefValue = redactStringValue(f.DefValue)
   388				}
   389			}
   390	
   391			return nil
   392		})
   393		if err != nil {
   394			return err
   395		}
   396	
   397		return nil
   398	}

   400	// GatherFlagsFromCobraCommand gathers the flags from the cobra command, and parses them according
   401	// to the field description passed in params. The result is a map of field
   402	// names to parsed values.
   403	//
   404	// If onlyProvided is true, only fields that are provided
   405	// by the user are returned (i.e. not the default values).
   406	//
   407	// If a field cannot be parsed correctly, or is missing even though it is not optional,
   408	// an error is returned.
   409	//
   410	// The required argument checks that all the required field definitions are present.
   411	// The provided argument only checks that the provided flags are passed.
   412	// Prefix is prepended to all flag names.
   413	func (pds *Definitions) GatherFlagsFromCobraCommand(
   414		cmd *cobra.Command,
   415		onlyProvided bool,
   416		ignoreRequired bool,
   417		prefix string,
   418		options ...ParseOption,
   419	) (*FieldValues, error) {
   420		ps := NewFieldValues()
   421	
   422		err := pds.ForEachE(func(pd *Definition) error {
   423			p := &FieldValue{
   424				Definition: pd,
   425			}
   426	
   427			if pd.IsArgument {
   428				return nil
   429			}
   430	
   431			// check if the flag is set
   432			flagName := prefix + pd.Name
   433			flagName = strings.ReplaceAll(flagName, "_", "-")
   434	
   435			if !cmd.Flags().Changed(flagName) {
   436				if pd.Required {
   437					if ignoreRequired {
   438						return nil
   439					}
   440	
   441					return errors.Errorf("Field %s is required", pd.Name)
   442				}
   443	
   444				if pd.Default == nil {
   445					return nil
   446				}
   447	
   448				if onlyProvided {
   449					return nil
   450				}
   451			}
   452	
   453			options := append([]ParseOption{
   454				WithMetadata(map[string]interface{}{
   455					"flag": flagName,
   456				}),
   457				WithSource("cobra"),
   458			}, options...)
   459	
   460			switch pd.Type {
   461			case TypeObjectFromFile,
   462				TypeObjectListFromFile,
   463				TypeStringFromFile,
   464				TypeStringListFromFile,
   465				TypeString,
   466				TypeSecret,
   467				TypeFile,
   468				TypeDate,
   469				TypeChoice:
   470				v, err := cmd.Flags().GetString(flagName)
   471				if err != nil {
   472					return err
   473				}
   474				v2, err := pd.ParseField([]string{v}, options...)
   475				if err != nil {
   476					return err
   477				}
   478				ps.Set(pd.Name, v2)
   479	
   480			case TypeFloat:
   481				v, err := cmd.Flags().GetFloat64(flagName)
   482				if err != nil {
   483					return err
   484				}
   485				err = p.Update(v, options...)
   486				if err != nil {
   487					return err
   488				}
   489				ps.Set(pd.Name, p)
   490	
   491			case TypeInteger:
   492				v, err := cmd.Flags().GetInt(flagName)
   493				if err != nil {
   494					return err
   495				}
   496				err = p.Update(v, options...)
   497				if err != nil {
   498					return err
   499				}
   500				ps.Set(pd.Name, p)
   501	
   502			case TypeBool:
   503				v, err := cmd.Flags().GetBool(flagName)
   504				if err != nil {
   505					return err
   506				}
   507				err = p.Update(v, options...)
   508				if err != nil {
   509					return err
   510				}
   511				ps.Set(pd.Name, p)
   512	
   513			case TypeObjectListFromFiles,
   514				TypeStringFromFiles,
   515				TypeStringListFromFiles,
   516				TypeFileList:
   517				v, err := cmd.Flags().GetStringSlice(flagName)
   518				if err != nil {
   519					return err
   520				}
   521				v2, err := pd.ParseField(v, options...)
   522				if err != nil {
   523					return err
   524				}
   525				ps.Set(pd.Name, v2)
   526	
   527			case TypeStringList,
   528				TypeChoiceList:
   529				v, err := cmd.Flags().GetStringSlice(flagName)
   530				if err != nil {
   531					return err
   532				}
   533				err = p.Update(v, options...)
   534				if err != nil {
   535					return err
   536				}
   537				ps.Set(pd.Name, p)
   538	
   539			case TypeKeyValue:
   540				v, err := cmd.Flags().GetStringSlice(flagName)
   541				if err != nil {
   542					return err
   543				}
   544	
   545				// if it was changed and is empty, then skip setting from default
   546				if cmd.Flags().Changed(flagName) && len(v) == 0 {
   547					options_ := append(options,
   548						WithMetadata(map[string]interface{}{
   549							"flag":      flagName,
   550							"emptyFlag": true,
   551						}))
   552					err = p.Update(map[string]string{}, options_...)
   553					if err != nil {
   554						return err
   555					}
   556					ps.Set(pd.Name, p)
   557				} else {
   558					v2, err := pd.ParseField(v, options...)
   559					if err != nil {
   560						return err
   561					}
   562					ps.Set(pd.Name, v2)
   563				}
   564	
   565			case TypeIntegerList:
   566				// NOTE(manuel, 2023-04-01) Do we not check for default here?
   567				v, err := cmd.Flags().GetIntSlice(flagName)
   568				if err != nil {
   569					return err
   570				}
   571				err = p.Update(v, options...)
   572				if err != nil {
   573					return err
   574				}
   575				ps.Set(pd.Name, p)
   576	
   577			case TypeFloatList:
   578				// NOTE(manuel, 2023-04-01) Do we not check for default here?
   579				v, err := cmd.Flags().GetFloat64Slice(flagName)
   580				if err != nil {
   581					return err
   582				}
   583				err = p.Update(v, options...)
   584				if err != nil {
   585					return err
   586				}
   587				ps.Set(pd.Name, p)
   588			}
   589	
   590			return nil
   591		})
   592		if err != nil {
   593			return nil, err
   594		}
   595		return ps, nil
   596	}
