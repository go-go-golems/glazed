    15	
    16	type cobraParserDemoSettings struct {
    17		ApiKey string `glazed:"api-key"`
    18	}
    19	
    20	func newCobraParserDemoSection(t *testing.T, defaultValue string) schema.Section {
    21		t.Helper()
    22		section, err := schema.NewSection(
    23			"demo",
    24			"Demo",
    25			schema.WithPrefix("demo-"),
    26			schema.WithFields(
    27				fields.New("api-key", fields.TypeString, fields.WithDefault(defaultValue)),
    28			),
    29		)
    30		require.NoError(t, err)
    31		return section
    32	}
    33	
    34	func executeParserForTest(t *testing.T, parser *CobraParser, args []string) *values.Values {
    35		t.Helper()
    36		var parsed *values.Values
    37		cmd := &cobra.Command{
    38			Use: "test",
    39			RunE: func(cmd *cobra.Command, args []string) error {
    40				var err error
    41				parsed, err = parser.Parse(cmd, args)
    42				return err
    43			},
    44		}
    45		require.NoError(t, parser.AddToCobraCommand(cmd))
    46		cmd.SetArgs(args)
    47		require.NoError(t, cmd.Execute())
    48		require.NotNil(t, parsed)
    49		return parsed
    50	}
    51	
    52	func TestCobraParserConfigPlanBuilderLoadsConfigFiles(t *testing.T) {
    53		section := newCobraParserDemoSection(t, "from-default")
    54		schema_ := schema.NewSchema(schema.WithSections(section))
    55	
    56		cfgFile := filepath.Join(t.TempDir(), "config.yaml")
    57		require.NoError(t, os.WriteFile(cfgFile, []byte("demo:\n  api-key: from-file\n"), 0o644))
    58	
    59		parser, err := NewCobraParserFromSections(schema_, &CobraParserConfig{
    60			SkipCommandSettingsSection: true,
    61			ConfigPlanBuilder: func(_ *values.Values, _ *cobra.Command, _ []string) (*glazedconfig.Plan, error) {
    62				return glazedconfig.NewPlan(
    63					glazedconfig.WithLayerOrder(glazedconfig.LayerExplicit),
    64				).Add(
    65					glazedconfig.ExplicitFile(cfgFile).Named("test-config"),
    66				), nil
    67			},
    68		})
    69		require.NoError(t, err)
    70	
    71		parsed := executeParserForTest(t, parser, nil)
    72		settings := &cobraParserDemoSettings{}
    73		require.NoError(t, parsed.DecodeSectionInto("demo", settings))
    74		require.Equal(t, "from-file", settings.ApiKey)
    75	}

    77	func TestCobraParserDoesNotImplicitlyLoadConfigFileWithoutPlanBuilder(t *testing.T) {
    78		section := newCobraParserDemoSection(t, "from-default")
    79		schema_ := schema.NewSchema(schema.WithSections(section))
    80	
    81		cfgFile := filepath.Join(t.TempDir(), "config.yaml")
    82		require.NoError(t, os.WriteFile(cfgFile, []byte("demo:\n  api-key: from-file\n"), 0o644))
    83	
    84		parser, err := NewCobraParserFromSections(schema_, &CobraParserConfig{
    85			AppName: "myapp",
    86		})
    87		require.NoError(t, err)
    88	
    89		parsed := executeParserForTest(t, parser, []string{"--config-file", cfgFile})
    90		settings := &cobraParserDemoSettings{}
    91		require.NoError(t, parsed.DecodeSectionInto("demo", settings))
    92		require.Equal(t, "from-default", settings.ApiKey)
    93	}
