    19	// CobraMiddlewaresFunc returns the full middleware chain for a Cobra command.
    20	// If you set it on CobraParserConfig, it replaces the built-in parser chain entirely.
    21	// That means you must re-add any sources you still want (for example env, config,
    22	// args, flags, or defaults) inside the custom function.
    23	type CobraMiddlewaresFunc func(
    24		parsedCommandSections *values.Values,
    25		cmd *cobra.Command,
    26		args []string,
    27	) ([]cmd_sources.Middleware, error)
    28	
    29	// CobraCommandDefaultMiddlewares builds a minimal middleware chain for Cobra commands.
    30	// It reads Cobra flags and positional arguments, then applies defaults.
    31	// Use it when you want an explicit chain; if you also need env loading or config
    32	// file loading, add those middlewares yourself.
    33	func CobraCommandDefaultMiddlewares(
    34		parsedCommandSections *values.Values,
    35		cmd *cobra.Command,
    36		args []string,
    37	) ([]cmd_sources.Middleware, error) {
    38		commandSettings := &CommandSettings{}
    39		err := parsedCommandSections.DecodeSectionInto(CommandSettingsSlug, commandSettings)
    40		if err != nil {
    41			return nil, err
    42		}
    43	
    44		// Default chain without legacy per-command file injection
    45		middlewares_ := []cmd_sources.Middleware{
    46			cmd_sources.FromCobra(cmd,
    47				fields.WithSource("cobra"),
    48			),
    49			cmd_sources.FromArgs(args,
    50				fields.WithSource("arguments"),
    51			),
    52		}
    53	
    54		middlewares_ = append(middlewares_,
    55			cmd_sources.FromDefaults(fields.WithSource(fields.SourceDefaults)),
    56		)
    57	
    58		return middlewares_, nil
    59	}

    61	// CobraParser takes a CommandDescription, and hooks it up to a cobra command.
    62	// It can then be used to parse the cobra flags and arguments back into a
    63	// set of SectionValues and a map[string]interface{} for the lose stuff.
    64	//
    65	// That command however doesn't have a Run* method, which is left to the caller to implement.
    66	//
    67	// This returns a CobraParser that can be used to parse the registered Sections
    68	// from the description.
    69	type CobraParser struct {
    70		Sections *schema.Schema
    71		// middlewaresFunc builds the parser chain after GlazedCommandSettings has been parsed.
    72		// At this point, Cobra has already parsed CLI flags and arguments, but the values
    73		// have not yet been turned into Glazed Values.
    74		//
    75		// In the built-in parser path, Glazed wires in env/config loading when configured.
    76		// If a custom function is supplied through CobraParserConfig, it replaces that
    77		// built-in chain and must re-add any sources it still needs.
    78		middlewaresFunc CobraMiddlewaresFunc
    79		// List of sections to be shown in short help, empty: always show all
    80		shortHelpSections []string
    81		// skipCommandSettingsSection controls whether the CommandSettingsSection should be automatically added
    82		skipCommandSettingsSection bool
    83		// enableProfileSettingsSection controls whether the ProfileSettingsSection should be added
    84		enableProfileSettingsSection bool
    85		// enableCreateCommandSettingsSection controls whether the CreateCommandSettingsSection should be added
    86		enableCreateCommandSettingsSection bool
    87	}

    89	type ConfigPlanBuilder func(parsedCommandSections *values.Values, cmd *cobra.Command, args []string) (*glazedConfig.Plan, error)
    90	
    91	type CobraParserConfig struct {
    92		// MiddlewaresFunc replaces the default Cobra parser chain.
    93		// Leave it nil to keep the built-in parser path, which can add env/config loading when configured.
    94		MiddlewaresFunc                    CobraMiddlewaresFunc
    95		ShortHelpSections                  []string
    96		SkipCommandSettingsSection         bool
    97		EnableProfileSettingsSection       bool
    98		EnableCreateCommandSettingsSection bool
    99		// AppName controls the default env prefix (strings.ToUpper(AppName)) used by the
   100		// built-in parser path. It enables env loading, but it does not imply config discovery.
   101		AppName string
   102		// ConfigPlanBuilder resolves explicit layered config policy for the command.
   103		// It is only used by the built-in parser path; if nil, no config files are loaded automatically.
   104		ConfigPlanBuilder ConfigPlanBuilder
   105	}

   107	func NewCobraCommandFromCommandDescription(
   108		description *cmds.CommandDescription,
   109	) *cobra.Command {
   110		cmd := &cobra.Command{
   111			Use:   description.Name,
   112			Short: description.Short,
   113			Long:  description.Long,
   114		}
   115		return cmd
   116	}

   118	// NewCobraParserFromSections creates a new CobraParser instance from a
   119	// CommandDescription, initializes the underlying cobra.Command, and adds all the
   120	// fields specified in the Sections CommandDescription to the cobra command.
   121	func NewCobraParserFromSections(
   122		sections *schema.Schema,
   123		cfg *CobraParserConfig,
   124	) (*CobraParser, error) {
   125		// Initialize parser with defaults
   126		ret := &CobraParser{
   127			Sections:                           sections,
   128			middlewaresFunc:                    CobraCommandDefaultMiddlewares,
   129			shortHelpSections:                  []string{},
   130			skipCommandSettingsSection:         false,
   131			enableProfileSettingsSection:       false,
   132			enableCreateCommandSettingsSection: false,
   133		}
   134		// Apply provided config if any
   135		if cfg != nil {
   136			if cfg.MiddlewaresFunc != nil {
   137				ret.middlewaresFunc = cfg.MiddlewaresFunc
   138			}
   139			ret.shortHelpSections = cfg.ShortHelpSections
   140			ret.skipCommandSettingsSection = cfg.SkipCommandSettingsSection
   141			ret.enableProfileSettingsSection = cfg.EnableProfileSettingsSection
   142			ret.enableCreateCommandSettingsSection = cfg.EnableCreateCommandSettingsSection
   143			// Build the built-in env/config-aware chain only when the caller did not supply a custom middleware function.
   144			if cfg.MiddlewaresFunc == nil {
   145				cfgCopy := *cfg
   146				ret.middlewaresFunc = func(parsedCommandSections *values.Values, cmd *cobra.Command, args []string) ([]cmd_sources.Middleware, error) {
   147					middlewares_ := []cmd_sources.Middleware{}
   148	
   149					// Append in reverse precedence so the last applied has highest precedence (flags)
   150					middlewares_ = append(middlewares_,
   151						cmd_sources.FromCobra(cmd,
   152							fields.WithSource("cobra"),
   153						),
   154					)
   155	
   156					middlewares_ = append(middlewares_,
   157						cmd_sources.FromArgs(args,
   158							fields.WithSource("arguments"),
   159						),
   160					)
   161	
   162					if cfgCopy.AppName != "" {
   163						envPrefix := strings.ToUpper(cfgCopy.AppName)
   164						middlewares_ = append(middlewares_,
   165							cmd_sources.FromEnv(envPrefix,
   166								fields.WithSource("env"),
   167							),
   168						)
   169					}
   170	
   171					if cfgCopy.ConfigPlanBuilder != nil {
   172						middlewares_ = append(middlewares_,
   173							cmd_sources.FromConfigPlanBuilder(
   174								func(_ context.Context, _ *values.Values) (*glazedConfig.Plan, error) {
   175									return cfgCopy.ConfigPlanBuilder(parsedCommandSections, cmd, args)
   176								},
   177								cmd_sources.WithParseOptions(fields.WithSource("config")),
   178							),
   179						)
   180					}
   181	
   182					middlewares_ = append(middlewares_,
   183						cmd_sources.FromDefaults(fields.WithSource(fields.SourceDefaults)),
   184					)
   185	
   186					return middlewares_, nil
   187				}
   188			}
   189		}
   190	
   191		// Only add the glazed command section if not explicitly skipped
   192		if !ret.skipCommandSettingsSection {
   193			commandSettingsSection, err := NewCommandSettingsSection()
   194			if err != nil {
   195				return nil, err
   196			}
   197			ret.Sections.Set(commandSettingsSection.GetSlug(), commandSettingsSection)
   198		}
   199	
   200		// Only add the profile settings section if explicitly enabled
   201		if ret.enableProfileSettingsSection {
   202			profileSettingsSection, err := NewProfileSettingsSection()
   203			if err != nil {
   204				return nil, err
   205			}
   206			ret.Sections.Set(profileSettingsSection.GetSlug(), profileSettingsSection)
   207		}
   208	
   209		// Only add the create command settings section if explicitly enabled
   210		if ret.enableCreateCommandSettingsSection {
   211			createCommandSettingsSection, err := NewCreateCommandSettingsSection()
   212			if err != nil {
   213				return nil, err
   214			}
   215			ret.Sections.Set(createCommandSettingsSection.GetSlug(), createCommandSettingsSection)
   216		}
   217	
   218		return ret, nil
   219	}

   221	func (c *CobraParser) AddToCobraCommand(cmd *cobra.Command) error {
   222		// NOTE(manuel, 2024-01-03) Maybe add some middleware functionality to whitelist/blacklist the Sections/fields that get added to the CLI
   223		// If we want to remove some fields from the CLI args (for example some output settings or so)
   224		err := c.Sections.ForEachE(func(_ string, section schema.Section) error {
   225			// check that section is a CobraSection
   226			// if not, return an error
   227			cobraSection, ok := section.(schema.CobraSection)
   228			if !ok {
   229				log.Error().Str("section", section.GetName()).Msg("Section is not a CobraSection")
   230				return errors.Errorf("section %s is not a CobraSection", section.GetName())
   231			}
   232	
   233			err := cobraSection.AddSectionToCobraCommand(cmd)
   234			if err != nil {
   235				log.Error().Err(err).Str("section", section.GetName()).Msg("Could not add section to cobra command")
   236				return err
   237			}
   238	
   239			return nil
   240		})
   241		if err != nil {
   242			return err
   243		}
   244	
   245		if len(c.shortHelpSections) > 0 {
   246			shortHelperSection := strings.Join(c.shortHelpSections, ",")
   247			cmd.Annotations["shortHelpSections"] = shortHelperSection
   248		}
   249	
   250		return nil
   251	}

   253	func (c *CobraParser) Parse(
   254		cmd *cobra.Command,
   255		args []string,
   256	) (*values.Values, error) {
   257		// We parse the glazed command settings first, since they will influence the following parsing
   258		// steps.
   259		parsedCommandSections, err := ParseCommandSettingsSection(cmd)
   260		if err != nil {
   261			return nil, err
   262		}
   263	
   264		// Create the middlewares by invoking the passed in middlewares constructor.
   265		// This is where applications can specify their own middlewares.
   266		middlewares_, err := c.middlewaresFunc(parsedCommandSections, cmd, args)
   267		if err != nil {
   268			return nil, err
   269		}
   270	
   271		parsedSections := values.New()
   272		err = cmd_sources.Execute(c.Sections, parsedSections, middlewares_...)
   273		if err != nil {
   274			return nil, err
   275		}
   276	
   277		return parsedSections, nil
   278	}

   280	// ParseGlazedCommandSection parses the global glazed settings from the given cobra.Command, if not nil,
   281	// and from the command itself if present.
   282	func ParseCommandSettingsSection(cmd *cobra.Command) (*values.Values, error) {
   283		parsedSections := values.New()
   284		commandSettingsSection, err := NewCommandSettingsSection()
   285		if err != nil {
   286			return nil, err
   287		}
   288	
   289		profileSettingsSection, err := NewProfileSettingsSection()
   290		if err != nil {
   291			return nil, err
   292		}
   293	
   294		createCommandSettingsSection, err := NewCreateCommandSettingsSection()
   295		if err != nil {
   296			return nil, err
   297		}
   298	
   299		commandSettingsSections := schema.NewSchema(schema.WithSections(
   300			commandSettingsSection,
   301			profileSettingsSection,
   302			createCommandSettingsSection,
   303		))
   304	
   305		// Parse the glazed command settings from the cobra command
   306		middlewares_ := []cmd_sources.Middleware{}
   307	
   308		if cmd != nil {
   309			middlewares_ = append(middlewares_, cmd_sources.FromCobra(cmd, fields.WithSource("cobra")))
   310		}
   311	
   312		err = cmd_sources.Execute(commandSettingsSections, parsedSections, middlewares_...)
   313		if err != nil {
   314			return nil, err
   315		}
   316	
   317		return parsedSections, nil
