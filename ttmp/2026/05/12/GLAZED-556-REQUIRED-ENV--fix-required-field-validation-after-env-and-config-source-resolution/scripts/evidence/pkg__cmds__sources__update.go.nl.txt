    12	// FromDefaults is a middleware that sets default values from field definitions.
    13	// It calls the next handler, and then iterates through each section and field definition.
    14	// If a default is defined, it sets that as the field value in the parsed values.
    15	func FromDefaults(options ...fields.ParseOption) Middleware {
    16		return func(next HandlerFunc) HandlerFunc {
    17			return func(schema_ *schema.Schema, parsedValues *values.Values) error {
    18				err := next(schema_, parsedValues)
    19				if err != nil {
    20					return err
    21				}
    22				err = schema_.UpdateWithDefaults(parsedValues, options...)
    23				if err != nil {
    24					return err
    25				}
    26				return nil
    27			}
    28		}
    29	}

    31	// FromMap takes a map where the keys are section slugs and the values are
    32	// maps of field name -> value. It calls next, and then merges the provided
    33	// values into the parsed values, skipping any sections not present in the schema.
    34	func FromMap(m map[string]map[string]interface{}, options ...fields.ParseOption) Middleware {
    35		return func(next HandlerFunc) HandlerFunc {
    36			return func(schema_ *schema.Schema, parsedValues *values.Values) error {
    37				err := next(schema_, parsedValues)
    38				if err != nil {
    39					return err
    40				}
    41	
    42				return updateFromMap(schema_, parsedValues, m, options...)
    43			}
    44		}
    45	}

    95	func updateFromMap(
    96		schema_ *schema.Schema,
    97		parsedValues *values.Values,
    98		m map[string]map[string]interface{},
    99		options ...fields.ParseOption) error {
   100		for k, v := range m {
   101			section, ok := schema_.Get(k)
   102			if !ok {
   103				continue
   104			}
   105	
   106			sectionValues := parsedValues.GetOrCreate(section)
   107			ps, err := section.GetDefinitions().GatherFieldsFromMap(v, true, options...)
   108			if err != nil {
   109				return err
   110			}
   111			_, err = sectionValues.Fields.Merge(ps)
   112			if err != nil {
   113				return err
   114			}
   115		}
   116		return nil
   117	}

   143	func updateFromEnv(
   144		schema_ *schema.Schema,
   145		parsedValues *values.Values,
   146		prefix string,
   147		options ...fields.ParseOption,
   148	) error {
   149		err := schema_.ForEachE(func(key string, l schema.Section) error {
   150			sectionValues := parsedValues.GetOrCreate(l)
   151			pds := l.GetDefinitions()
   152			sectionPrefix := l.GetPrefix()
   153			err := pds.ForEachE(func(p *fields.Definition) error {
   154				// Compute env key based on section prefix + field name, hyphen->underscore, uppercase,
   155				// and optional global prefix (app name) separated by underscore.
   156				base := sectionPrefix + p.Name
   157				envKey := strings.ToUpper(strings.ReplaceAll(base, "-", "_"))
   158				if prefix != "" {
   159					envKey = strings.ToUpper(prefix) + "_" + envKey
   160				}
   161	
   162				if v, ok := os.LookupEnv(envKey); ok {
   163					opts := append([]fields.ParseOption{
   164						fields.WithMetadata(map[string]interface{}{
   165							"env_key": envKey,
   166						}),
   167						fields.WithSource("env"),
   168					}, options...)
   169	
   170					// Parse env string into the appropriate typed value using the field's parser.
   171					// For list-like types, split on commas (trim brackets and whitespace).
   172					var inputs []string
   173					if p.Type.IsList() {
   174						s := strings.TrimSpace(v)
   175						// Trim optional surrounding brackets: [a,b,c]
   176						if strings.HasPrefix(s, "[") && strings.HasSuffix(s, "]") {
   177							s = strings.TrimPrefix(s, "[")
   178							s = strings.TrimSuffix(s, "]")
   179							s = strings.TrimSpace(s)
   180						}
   181						if s == "" {
   182							inputs = []string{}
   183						} else {
   184							parts := strings.Split(s, ",")
   185							for _, part := range parts {
   186								inputs = append(inputs, strings.TrimSpace(part))
   187							}
   188						}
   189					} else {
   190						inputs = []string{v}
   191					}
   192	
   193					pp, err := p.ParseField(inputs, opts...)
   194					if err != nil {
   195						return err
   196					}
   197					// Preserve parse log/metadata when updating parsed fields.
   198					if err := sectionValues.Fields.UpdateWithLog(p.Name, p, pp.Value, pp.Log...); err != nil {
   199						return err
   200					}
   201				}
   202				return nil
   203			})
   204			if err != nil {
   205				return err
   206			}
   207			return nil
   208		})
   209	
   210		return err
   211	}

   213	func FromEnv(prefix string, options ...fields.ParseOption) Middleware {
   214		return func(next HandlerFunc) HandlerFunc {
   215			return func(schema_ *schema.Schema, parsedValues *values.Values) error {
   216				err := next(schema_, parsedValues)
   217				if err != nil {
   218					return err
   219				}
   220	
   221				return updateFromEnv(schema_, parsedValues, prefix, options...)
   222			}
   223		}
   224	}

   226	func updateFromStringList(schema_ *schema.Schema, parsedValues *values.Values, prefix string, args []string, options ...fields.ParseOption) error {
   227		err := schema_.ForEachE(func(key string, l schema.Section) error {
   228			sectionValues := parsedValues.GetOrCreate(l)
   229			pds := l.GetDefinitions()
   230			ps, remainingArgs, err := pds.GatherFlagsFromStringList(args, true, true, prefix, options...)
   231			if err != nil {
   232				return err
   233			}
   234	
   235			_, err = sectionValues.Fields.Merge(ps)
   236			if err != nil {
   237				return err
   238			}
   239	
   240			ps, err = pds.GatherArguments(remainingArgs, true, true, options...)
   241			if err != nil {
   242				return err
   243			}
   244	
   245			_, err = sectionValues.Fields.Merge(ps)
   246			if err != nil {
   247				return err
   248			}
   249	
   250			return nil
   251		})
   252	
   253		if err != nil {
   254			return err
   255		}
   256	
   257		return nil
   258	}
